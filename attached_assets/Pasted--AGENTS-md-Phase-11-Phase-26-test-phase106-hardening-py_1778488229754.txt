修正方式：
python# 在 evaluate_form4_insider_signal 裡加入
def _is_likely_systematic_plan(items: list[dict]) -> bool:
    if len(items) < 4:
        return False
    dates = sorted(item.get("transaction_date", "") for item in items)
    # 如果日期分佈均勻（每月都有）且金額相似，視為計畫性賣出
    if len(set(d[:7] for d in dates if d)) >= len(dates) * 0.6:
        values = [item.get("value", 0) or 0 for item in items]
        avg = sum(values) / len(values)
        if all(abs(v - avg) / avg < 0.5 for v in values if avg > 0):
            return True
    return False

# 在分數計算前加
likely_systematic = _is_likely_systematic_plan(disposition_items)
if likely_systematic:
    disposition_weight = 0.3  # 降低計畫性賣出的負面權重
    # 加入說明
    limitations.append(
        "Disposition pattern suggests possible systematic plan (e.g. 10b5-1). "
        "Negative weight reduced. Requires human verification."
    )
